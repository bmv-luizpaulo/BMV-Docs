# NextAuth Configuration
NEXTAUTH_URL=http://localhost:9002
NEXTAUTH_SECRET=bmv-docs-secret-key-2024-development-only

# Google OAuth Configuration
# IMPORTANTE: Você precisa configurar essas credenciais no Google Cloud Console
# 1. Acesse: https://console.developers.google.com/
# 2. Crie um projeto ou selecione um existente
# 3. Ative a API do Google+ e Google OAuth2
# 4. Vá para "Credenciais" e crie uma credencial OAuth 2.0
# 5. Configure as URLs autorizadas:
#    - URIs de redirecionamento: http://localhost:9002/api/auth/callback/google
#    - Origens JavaScript: http://localhost:9002
GOOGLE_CLIENT_ID=your-google-client-id-here
GOOGLE_CLIENT_SECRET=your-google-client-secret-here

# Firebase Configuration
# IMPORTANTE: Obtenha essas credenciais no Firebase Console
# 1. Acesse: https://console.firebase.google.com/
# 2. Selecione seu projeto
# 3. Vá para "Configurações do projeto" > "Seus aplicativos"
# 4. Adicione um aplicativo web e copie as configurações
FIREBASE_API_KEY=your-firebase-api-key
FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com
FIREBASE_PROJECT_ID=your-project-id
FIREBASE_STORAGE_BUCKET=your-project.appspot.com
FIREBASE_MESSAGING_SENDER_ID=your-sender-id
FIREBASE_APP_ID=your-app-id

# Google AI (Gemini) Configuration
# IMPORTANTE: Obtenha esta chave no Google AI Studio
# 1. Acesse: https://aistudio.google.com/
# 2. Clique em "Get API key" e copie a chave.
# 3. Adicione esta chave também nas Environment Variables do seu projeto na Vercel.
GEMINI_API_KEY=your-gemini-api-key-here


# Ambiente
NODE_ENV=development
